179  int AtInterface(const CVList& cvs)
180  {
181  std::vector<double> dists;
182  dists.resize(_centers.size());
183 
184  // Record the difference between all cvs and all nodes
185  for (size_t i = 0; i < _centers.size(); i++)
186  dists[i] = (cvs[0]->GetValue() - _centers[i])*(cvs[0]->GetValue() - _centers[i]);
187 
188  return (std::min_element(dists.begin(), dists.end()) - dists.begin());
189  }
