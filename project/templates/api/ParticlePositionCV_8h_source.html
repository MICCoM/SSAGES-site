101  void Evaluate(const Snapshot& snapshot) override
102  {
103  // Get local atom indices and compute COM.
104  std::vector<int> idx;
105  snapshot.GetLocalIndices(_atomids, &idx);
106 
107  // Get data from snapshot.
108  auto n = snapshot.GetNumAtoms();
109  const auto& masses = snapshot.GetMasses();
110 
111  // Initialize gradient.
112  std::fill(_grad.begin(), _grad.end(), Vector3{0,0,0});
113  _grad.resize(n, Vector3{0,0,0});
114 
115  // Compute total and center of mass.
116  auto masstot = snapshot.TotalMass(idx);
117  Vector3 com = snapshot.CenterOfMass(idx, masstot);
118 
119  // Compute difference between point and account for requested
120  // dimensions only.
121  Vector3 dx = snapshot.ApplyMinimumImage(com - _point).cwiseProduct(_fix.cast<double>());
122  _val = dx.norm();
123 
124  // If distance is zero, we have nothing to do.
125  if(_val == 0)
126  return;
127 
128  for(auto& id : idx)
129  _grad[id] = dx/_val*masses[id]/masstot;
130  }
