425  Vector3 CenterOfMass(const Label& indices, double mtot) const
426  {
427  // Store coorinates and masses in vectors to gather.
428  std::vector<double> pos, mass, gpos, gmass;
429  std::vector<int> pcounts(_comm.size(), 0), mcounts(_comm.size(), 0);
430  std::vector<int> pdispls(_comm.size()+1, 0), mdispls(_comm.size()+1, 0);
431 
432  pcounts[_comm.rank()] = 3*indices.size();
433  mcounts[_comm.rank()] = indices.size();
434 
435  // Reduce counts.
436  MPI_Allreduce(MPI_IN_PLACE, pcounts.data(), pcounts.size(), MPI_INT, MPI_SUM, _comm);
437  MPI_Allreduce(MPI_IN_PLACE, mcounts.data(), mcounts.size(), MPI_INT, MPI_SUM, _comm);
438 
439  // Compute displacements.
440  std::partial_sum(pcounts.begin(), pcounts.end(), pdispls.begin() + 1);
441  std::partial_sum(mcounts.begin(), mcounts.end(), mdispls.begin() + 1);
442 
443  // Fill up mass and position vectors.
444  for(auto& idx : indices)
445  {
446  auto& p = _positions[idx];
447  pos.push_back(p[0]);
448  pos.push_back(p[1]);
449  pos.push_back(p[2]);
450  mass.push_back(_masses[idx]);
451  }
452 
453  // Re-size receiving vectors.
454  gpos.resize(pdispls.back(), 0);
455  gmass.resize(mdispls.back(), 0);
456 
457  // All-gather data.
458  MPI_Allgatherv(pos.data(), pos.size(), MPI_DOUBLE, gpos.data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, _comm);
459  MPI_Allgatherv(mass.data(), mass.size(), MPI_DOUBLE, gmass.data(), mcounts.data(), mdispls.data(), MPI_DOUBLE, _comm);
460 
461  // Loop through atoms and compute mass weighted sum.
462  // We march linearly through list and find nearest image
463  // to each successive particle to properly unwrap object.
464  Vector3 ppos = {gpos[0], gpos[1], gpos[2]}; // Previous unwrapped position.
465  Vector3 cpos = ppos;
466  Vector3 xcm = gmass[0]*cpos;
467 
468  for(size_t i = 1, j = 3; i < gmass.size(); ++i, j += 3)
469  {
470  cpos = {gpos[j], gpos[j+1], gpos[j+2]};
471  cpos = ApplyMinimumImage(cpos - ppos) + ppos;
472  xcm += gmass[i]*cpos;
473  ppos = cpos;
474  }
475 
476  return xcm/mtot;
477  }
