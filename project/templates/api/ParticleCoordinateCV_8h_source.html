98

102 
103  // Initialize gradient.
104  std::fill(_grad.begin(), _grad.end(), Vector3{0,0,0});
105  _grad.resize(n, Vector3{0,0,0});
106 
107  // Compute total and center of mass.
108  auto masstot = snapshot.TotalMass(idx);
109  Vector3 com = snapshot.CenterOfMass(idx, masstot);
110 
111  // Assign CV value.
112  switch(_dim)
113  {
114  case Dimension::x:
115  _val = com[0];
116  break;
117  case Dimension::y:
118  _val = com[1];
119  break;
120  case Dimension::z:
121  _val = com[2];
122  break;
123  }
124 
125  // Assign gradient to appropriate atoms.
126  for(auto& id : idx)
127  {
128  _grad[id][0] = (_dim == Dimension::x) ? masses[id]/masstot : 0;
129  _grad[id][1] = (_dim == Dimension::y) ? masses[id]/masstot : 0;
130  _grad[id][2] = (_dim == Dimension::z) ? masses[id]/masstot : 0;
