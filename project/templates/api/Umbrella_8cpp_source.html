47  void Umbrella::PostIntegration(Snapshot* snapshot, const CVList& cvs)
48  {
49  // Compute the forces on the atoms from the CV's using the chain
50  // rule.
51  auto& forces = snapshot->GetForces();
52  for(size_t i = 0; i < cvs.size(); ++i)
53  {
54  // Get current CV and gradient.
55  auto& cv = cvs[i];
56  auto& grad = cv->GetGradient();
57 
58  // Compute dV/dCV.
59  auto D = _kspring[i]*(cv->GetDifference(_centers[i]));
60 
61  // Update forces.
62  for(size_t j = 0; j < forces.size(); ++j)
63  for(size_t k = 0; k < forces[j].size(); ++k)
64  forces[j][k] -= D*grad[j][k];
65  }
66

68  if(_iteration % _logevery == 0)
69  PrintUmbrella(cvs);
70  }
