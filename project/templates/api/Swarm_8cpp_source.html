34  bool Swarm::CVInitialized(const CVList& cvs)
35  {
36  double threshold = 0.2;
37  const double eps = 0.0000000001;
38  double diff;
39 
40  //On the first iteration, check that the CVs are within (threshold*100)% of the center value they're associated to
41  for(size_t i = 0; i < cvs.size(); i++)
42  {
43  if(centers_[i] <= eps)
44  {//e.g. if centers_[i] = 0
45  diff = std::abs((cvs[i]->GetValue() - (centers_[i]+eps)) / ((cvs[i]->GetValue() + (eps + centers_[i]))/2.0));
46  }
47  else
48  {
49  diff = std::abs((cvs[i]->GetValue() - (centers_[i])) / ((cvs[i]->GetValue() + (centers_[i]))/2.0));
50  }
51  if(diff >= threshold)
52  {
53  return false; //proceed to initialize again
54  }
55  }
56  return true; //OK to move on to regular sampling
57  }

90  //Do restrained sampling, and do not harvest trajectories
91  if(!original_initialized)
92  {

94  {

96  auto& cv = cvs[i];
97  auto& grad = cv->GetGradient();

129  velocities[localindex][1] = prev_velocities_[index_][i*3 + 1];
130  velocities[localindex][2] = prev_velocities_[index_][i*3 + 2];
131 
132  }
133  }
134  }
135  }
136  else
137  {
138  if(!sampling_started)

141  }

197

199  }

202  }

207  {
208  //End of trajectory, harvest drift
209  for(size_t i = 0; i < cv_drift_.size(); i++)
210  {

212  }
