98  void Meta::AddHill(const CVList& cvs)
99  {
100  int n = cvs.size();
101 
102  // Assume we have the same number of procs per walker.
103  int nwalkers = _world.size()/_comm.size();
104 
105  // We need to exchange CV values across the walkers
106  // and to each proc on a walker.
107  std::vector<double> cvals(n*nwalkers, 0);
108 
109  if(_comm.rank() == 0)
110  {
111  for(auto i = 0, j = _world.rank()/_comm.size()*n; i < n; ++i,++j)
112  cvals[j] = cvs[i]->GetValue();
113  }
114 
115  // Reduce across all processors and add hills.
116  MPI_Allreduce(MPI_IN_PLACE, cvals.data(), n*nwalkers, MPI_DOUBLE, MPI_SUM, _world);
117 
118  for(int i = 0; i < n*nwalkers; i += n)
119  {
120  std::vector<double> cval(cvals.begin() + i, cvals.begin() + i + n);
121  _hills.emplace_back(cval, _widths, _height);
122 
123  // Write hill to file.
124  if(_world.rank() == 0)
125  PrintHill(_hills.back());
126  }
127  }

142  void Meta::CalcBiasForce(const CVList& cvs)
143  {
144  // Reset bias and derivatives.
145  _bias = 0;
146  std::fill(_derivatives.begin(), _derivatives.end(), 0);
147 
148  // Loop through hills and calculate the bias force.
149  for(auto& hill : _hills)
150  {
151  auto n = hill.center.size();
152  std::vector<double> tder(n, 1.0), dx(n, 1);
153  auto tbias = 1.;
154 
155  // Initialize dx and tbias.
156  for(size_t i = 0; i < n; ++i)
157  {
158  dx[i] = cvs[i]->GetDifference(hill.center[i]);
159  tbias *= gaussian(dx[i], hill.width[i]);
160  }
161 
162  for(size_t i = 0; i < n; ++i)
163  for(size_t j = 0; j < n; ++j)
164  {
165  if(j != i)
166  tder[i] *= gaussian(dx[j], hill.width[j]);
167  else
168  tder[i] *= gaussianDerv(dx[j], hill.width[j]);
169  }
170 
171  _bias += _height * tbias;
172  for(size_t i = 0; i < n; ++i)
173  _derivatives[i] += _height*tder[i];
174  }
175  }
