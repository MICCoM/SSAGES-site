91  // Post-integration hook.
92  void Meta::PostIntegration(Snapshot* snapshot, const CVList& cvs)
93  {
94  // Add hills when needed.
95  if(snapshot->GetIteration() % hillfreq_ == 0)
96  AddHill(cvs, snapshot->GetIteration());
97 
98  // Always calculate the current bias.
99  CalcBiasForce(cvs);
100 
101  // Take each CV and add its biased forces to the atoms
102  // using the chain rule.
103  auto& forces = snapshot->GetForces();
104  auto& virial = snapshot->GetVirial();
105 
106  for(size_t i = 0; i < cvs.size(); ++i)
107  {
108  auto& grad = cvs[i]->GetGradient();
109  auto& boxgrad = cvs[i]->GetBoxGradient();
110 
111  // Update the forces in snapshot by adding in the force bias from each
112  // CV to each atom based on the gradient of the CV.
113  for (size_t j = 0; j < forces.size(); ++j)
114  for(size_t k = 0; k < 3; ++k)
115  forces[j][k] -= derivatives_[i]*grad[j][k];
116 
117  virial += derivatives_[i]*boxgrad;
118  }
119  }

158  // Drop a new hill.
159  void Meta::AddHill(const CVList& cvs, int iteration)
160  {
161  int n = cvs.size();
162 
163  // Assume we have the same number of procs per walker.
164  int nwalkers = world_.size()/comm_.size();
165 
166  // We need to exchange CV values across the walkers
167  // and to each proc on a walker.
168  std::vector<double> cvals(n*nwalkers, 0);
169 
170  if(comm_.rank() == 0)
171  {
172  for(auto i = 0, j = world_.rank()/comm_.size()*n; i < n; ++i,++j)
173  cvals[j] = cvs[i]->GetValue();
174  }
175 
176  // Reduce across all processors and add hills.
177  MPI_Allreduce(MPI_IN_PLACE, cvals.data(), n*nwalkers, MPI_DOUBLE, MPI_SUM, world_);
178 
179  for(int i = 0; i < n*nwalkers; i += n)
180  {
181  std::vector<double> cval(cvals.begin() + i, cvals.begin() + i + n);
182  hills_.emplace_back(cval, widths_, height_);
183 
184  // Write hill to file.
185  if(world_.rank() == 0)
186  PrintHill(hills_.back(), iteration);
187  }
188 
189  // If grid is defined, add bias onto grid.
190  if(grid_ != nullptr)
191  {
192  std::vector<double> dx(n, 0.0), df(n, 1.0);
193  auto& hill = hills_.back();
194  for(auto it = grid_->begin(); it != grid_->end(); ++it)
195  {
196  auto& val = *it;
197  auto coord = it.coordinates();
198 
199  // Compute difference between grid point and current val.
200  for(size_t i = 0; i < n; ++i)
201  {
202  dx[i] = -cvs[i]->GetDifference(coord[i]);
203  df[i] = 1.;
204  }
205 
206  // Compute derivative.
207  for(size_t i = 0; i < n; ++i)
208  {
209  for(size_t j = 0; j < n; ++j)
210  {
211  if(j != i)
212  df[i] *= gaussian(dx[j], hill.width[j]);
213  else
214  df[i] *= gaussianDerv(dx[j], hill.width[j]);
215  }
216  }
217 
218  // Add to grid.
219  for(size_t i = 0; i < n; ++i)
220  val[i] += height_*df[i];
221  }
222  }
223  }

244  void Meta::CalcBiasForce(const CVList& cvs)
245  {
246  // Reset bias and derivatives.
247  double bias = 0.;
248  auto n = cvs.size();
249 
250  // Reset vectors.
251  std::fill(derivatives_.begin(), derivatives_.end(), 0);
252 
253  // Look up and apply grid bias.
254  if(grid_ != nullptr)
255  {
256  bool inbounds = true;
257  std::vector<double> val(n, 0.);
258  for(size_t i = 0; i < n; ++i)
259  {
260  val[i] = cvs[i]->GetValue();
261  if(val[i] < grid_->GetLower(i) || val[i] > grid_->GetUpper(i))
262  inbounds = false;
263  }
264 
265  if(inbounds)
266  {
267  auto frc = (*grid_)[val];
268  for(size_t i = 0; i < n; ++i)
269  derivatives_[i] = frc[i];
270  }
271  else
272  {
273  if(comm_.rank() == 0)
274  {
275  std::cerr << "Metadynamics: out of bounds ( ";
276  for(auto& v : val)
277  std::cerr << v << " ";
278  std::cerr << ")" << std::endl;
279  }
280  }
281  }
282  else
283  {
284  // Loop through hills and calculate the bias force.
285  for(auto& hill : hills_)
286  {
287  auto tbias = 1.;
288  std::fill(tder_.begin(), tder_.end(), 1.0);
289  std::fill(dx_.begin(), dx_.end(), 1.0);
290 
291  for(size_t i = 0; i < n; ++i)
292  {
293  dx_[i] = cvs[i]->GetDifference(hill.center[i]);
294  tbias *= gaussian(dx_[i], hill.width[i]);
295  }
296 
297  for(size_t i = 0; i < n; ++i)
298  for(size_t j = 0; j < n; ++j)
299  {
300  if(j != i)
301  tder_[i] *= gaussian(dx_[j], hill.width[j]);
302  else
303  tder_[i] *= gaussianDerv(dx_[j], hill.width[j]);
304  }
305 
306  bias += height_ * tbias;
307  for(size_t i = 0; i < n; ++i)
308  derivatives_[i] += height_*tder_[i];
309  }
310  }
311 
312  // Restraints.
313  for(size_t i = 0; i < n; ++i)
314  {
315  auto cval = cvs[i]->GetValue();
316  if(cval < lowerb_[i])
317  derivatives_[i] += lowerk_[i]*(cval - lowerb_[i]);
318  else if(cval > upperb_[i])
319  derivatives_[i] += upperk_[i]*(cval - upperb_[i]);
320  }
321  }
322 }
