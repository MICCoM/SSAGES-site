143  void StringReparam(double alpha_star)
144  {
145  std::vector<double> alpha_star_vector(_numnodes,0.0);
146 
147  //Reparameterization
148  //Alpha star is the uneven mesh, approximated as linear distance between points
149  if(_comm.rank()==0)
150  alpha_star_vector[_mpiid] = _mpiid == 0 ? 0 : alpha_star;
151 
152  //Gather each alpha_star into a vector
153  MPI_Allreduce(MPI::IN_PLACE, &alpha_star_vector[0], _numnodes, MPI::DOUBLE, MPI::SUM, _world);
154 
155  for(size_t i = 1; i < alpha_star_vector.size(); i++)
156  alpha_star_vector[i] += alpha_star_vector[i-1];
157 
158  for(size_t i = 1; i < alpha_star_vector.size(); i++)
159  alpha_star_vector[i] /= alpha_star_vector[_numnodes - 1];
160 
161  tk::spline spl; //Cubic spline interpolation
162 
163  for(size_t i = 0; i < _centers.size(); i++)
164  {
165  std::vector<double> cvs_new(_numnodes, 0.0);
166 
167  if(_comm.rank() == 0)
168  cvs_new[_mpiid] = _centers[i];
169 
170  MPI_Allreduce(MPI::IN_PLACE, &cvs_new[0], _numnodes, MPI::DOUBLE, MPI::SUM, _world);
171 
172  spl.set_points(alpha_star_vector, cvs_new);
173  _centers[i] = spl(_mpiid/(_numnodes - 1.0));
174  }
175  }

241  // StoreSnapShot and SetSnapshot should be put in the snapshot routine.
242  // Furthermore, there can be some efficiency gains made by storing the
243  // local snapshot, and when you need to set the snapshot,
244  // that is when you serialize.
245  void StoreSnapshot(Snapshot* snapshot, int frame = 0)
246  {
247  auto locatoms = snapshot->GetNumAtoms();
248  const auto& Pos = snapshot->GetPositions();
249  const auto& Vel = snapshot->GetVelocities();
250  const auto& IDs = snapshot->GetAtomIDs();
251 
252  std::vector<int> pcounts(_comm.size(), 0), mcounts(_comm.size(), 0);
253  std::vector<int> pdispls(_comm.size()+1, 0), mdispls(_comm.size()+1, 0);
254 
255  pcounts[_comm.rank()] = 3*locatoms;
256  mcounts[_comm.rank()] = locatoms;
257 
258  // Reduce counts.
259  MPI_Allreduce(MPI_IN_PLACE, pcounts.data(), pcounts.size(), MPI_INT, MPI_SUM, _comm);
260  MPI_Allreduce(MPI_IN_PLACE, mcounts.data(), mcounts.size(), MPI_INT, MPI_SUM, _comm);
261 
262  // Compute displacements.
263  std::partial_sum(pcounts.begin(), pcounts.end(), pdispls.begin() + 1);
264  std::partial_sum(mcounts.begin(), mcounts.end(), mdispls.begin() + 1);
265 
266  // Re-size receiving vectors.
267  _prev_positions[frame].resize(pdispls.back(), 0);
268  _prev_velocities[frame].resize(pdispls.back(), 0);
269  _prev_IDs[frame].resize(mdispls.back(), 0);
270 
271  std::vector<double> ptemp;
272  std::vector<double> vtemp;
273 
274  for(auto& p : Pos)
275  {
276  ptemp.push_back(p[0]);
277  ptemp.push_back(p[1]);
278  ptemp.push_back(p[2]);
279  }
280 
281  for(auto& v : Vel)
282  {
283  vtemp.push_back(v[0]);
284  vtemp.push_back(v[1]);
285  vtemp.push_back(v[2]);
286  }
287 
288  // All-gather data.
289  MPI_Allgatherv(ptemp.data(), ptemp.size(), MPI_DOUBLE, _prev_positions[frame].data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, _comm);
290  MPI_Allgatherv(vtemp.data(), vtemp.size(), MPI_DOUBLE, _prev_velocities[frame].data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, _comm);
291  MPI_Allgatherv(IDs.data(), IDs.size(), MPI_INT, _prev_IDs[frame].data(), mcounts.data(), mdispls.data(), MPI_INT, _comm);
292  }
