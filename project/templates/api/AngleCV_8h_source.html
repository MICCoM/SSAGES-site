84  void Evaluate(const Snapshot& snapshot) override
85  {
86  // Get data from snapshot.
87  auto n = snapshot.GetNumAtoms();
88  const auto& pos = snapshot.GetPositions();
89  auto& comm = snapshot.GetCommunicator();
90 
91  // Initialize gradient.
92  std::fill(_grad.begin(), _grad.end(), Vector3{0,0,0});
93  _grad.resize(n, Vector3{0,0,0});
94 
95  Vector3 xi{0, 0, 0}, xj{0, 0, 0}, xk{0, 0, 0};
96 
97  auto iindex = snapshot.GetLocalIndex(_atomids[0]);
98  auto jindex = snapshot.GetLocalIndex(_atomids[1]);
99  auto kindex = snapshot.GetLocalIndex(_atomids[2]);
100 
101  if(iindex != -1) xi = pos[iindex];
102  if(jindex != -1) xj = pos[jindex];
103  if(kindex != -1) xk = pos[kindex];
104 
105  // By performing a reduce, we actually collect all. This can
106  // be converted to a more intelligent allgater on rank then bcast.
107  MPI_Allreduce(MPI_IN_PLACE, xi.data(), 3, MPI_DOUBLE, MPI_SUM, comm);
108  MPI_Allreduce(MPI_IN_PLACE, xj.data(), 3, MPI_DOUBLE, MPI_SUM, comm);
109  MPI_Allreduce(MPI_IN_PLACE, xk.data(), 3, MPI_DOUBLE, MPI_SUM, comm);
110 
111  // Two vectors
112  auto rij = snapshot.ApplyMinimumImage(xi - xj);
113  auto rkj = snapshot.ApplyMinimumImage(xk - xj);
114 
115  auto dotP = rij.dot(rkj);
116  auto nrij = rij.norm();
117  auto nrkj = rkj.norm();
118 
119  _val = acos(dotP/(nrij*nrkj));
120 
121  // Calculate gradients
122  auto prefactor = -1.0/(sqrt(1. - dotP/(nrij*nrkj))*nrij*nrkj);
123 
124  Vector3 gradi{0, 0, 0}, gradk{0, 0, 0};
125  if(iindex != -1) gradi = prefactor*(rkj - dotP*rij/(nrij*nrij));
126  if(kindex != -1) gradk = prefactor*(rij - dotP*rkj/(nrkj*nrkj));
127  MPI_Allreduce(MPI_IN_PLACE, gradi.data(), 3, MPI_DOUBLE, MPI_SUM, comm);
128  MPI_Allreduce(MPI_IN_PLACE, gradk.data(), 3, MPI_DOUBLE, MPI_SUM, comm);
129  if(iindex != -1) _grad[iindex] = gradi;
130  if(kindex != -1) _grad[kindex] = gradk;
131  if(jindex != -1) _grad[jindex] = -gradi - gradk;
132  }
