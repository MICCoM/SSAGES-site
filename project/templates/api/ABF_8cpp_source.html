35  // Function to return histogram coordinates, given CV values. Returns -1 if any CV value is outside bounds.
36  // Regardless of number of CVs, the histogram is a one dimensional object.
37  // Therefore, this function works in two parts:
38  // 1) Mapping from CV coordinates to an N dimensional fictitious object where N is the number of CVs.
39  // -> This fictitious object is (Number of CV bins) large in each dimension.
40  // -> Example: If you have two CVs, X and Y, and you bin X into three bins and Y into five bins, the fictitious object is a 2 dimensional 3x5 matrix.
41 
42  // 2) Mapping from N dimensions to one dimension.
43  // -> Simply constructed by writing out all members of the fictitious object in a line. This is a vector of length equal to the product of number of bins in each dimension.
44  // -> For the above example, this is 3.5 = 15 members. First 5 members are Y = [1 2 3 4 5] bins with X = 1. Next 5 members are Y = [1 2 3 4 5] with X = 2 ....

57  int ABF::histCoords(const CVList& cvs)
58  {
59  // Histogram details: This is a 2 Dimensional object set up as the following:
60  // Histdetails is a vector of three vectors, each of those three vectors are (Number of CVs) long.
61  // First of these vectors hold the lower bound for the CVs in order.
62  // Second vector holds the upper bound for the CVs in order.
63  // Third vector holds number of histogram bins for the CVs in order.
64 
65  // The first mapping starts here.
66  for(size_t i = 0; i < _histdetails.size(); ++i)
67  {
68  // Check if CV is in bounds for each CV.
69  if ((cvs[i]->GetValue()<_histdetails[i][0]) || (cvs[i]->GetValue()>_histdetails[i][1]))
70  return -1;
71  }
72 
73  // This vector holds the CV coordinates in the fictitious object.
74  std::vector<int> coords(cvs.size());
75 
76  // Loop over each CV dimension.
77  for(size_t i = 0; i < _histdetails.size(); ++i)
78  {
79  // Loop over all bins in each CV dimensions
80  for(int j = 0; j < _histdetails[i][2] ; ++j)
81  {
82  if((_histdetails[i][0] + j*((_histdetails[i][1]-_histdetails[i][0])/_histdetails[i][2]) < cvs[i]->GetValue()) &&
83  (_histdetails[i][0] + (j+1)*((_histdetails[i][1]-_histdetails[i][0])/_histdetails[i][2]) > cvs[i]->GetValue()))
84  {
85  coords[i] = j;
86  }
87  }
88  } //First mapping ends here.
89 
90  // Second mapping starts here.
91  int finalcoord = 0;
92  int indexer = 1;
93  // Loop over each CV dimension.
94  for(size_t i = 0; i < _histdetails.size(); ++i)
95  {
96  indexer = 1;
97  for(size_t j = i+1; j < _histdetails.size(); ++j)
98  indexer = indexer*_histdetails[j][2];
99 
100  finalcoord += coords[i]*indexer;
101  }
102 
103  return finalcoord;
104  }

153  void ABF::PostIntegration(Snapshot* snapshot, const CVList& cvs)
154  {
155  ++_iteration;
156 
157  // Gather information.
158  auto& vels = snapshot->GetVelocities();
159  auto& mass = snapshot->GetMasses();
160  auto& forces = snapshot->GetForces();
161  auto n = snapshot->GetNumAtoms();
162 
164  int coord = histCoords(cvs);
165 
167  Eigen::MatrixXd J(_ncv, 3*n);
168 
169  // Fill J. Each column represents grad(CV) with flattened Cartesian elements.
170  for(int i = 0; i < _ncv; ++i)
171  {
172  auto& grad = cvs[i]->GetGradient();
173  for(size_t j = 0; j < n; ++j)
174  J.block<1, 3>(i,3*j) = grad[j];
175  }
176 
177  //* Calculate W using Darve's approach (http://mc.stanford.edu/cgi-bin/images/0/06/Darve_2008.pdf).
178  Eigen::MatrixXd Jmass = J.transpose();
179 
180  for(size_t i = 0; i < forces.size(); ++i)
181  Jmass.block(3*i, 0, 3, _ncv) = Jmass.block(3*i, 0, 3, _ncv)/mass[i];
182 
183  Eigen::MatrixXd Minv = J*Jmass;
184  MPI_Allreduce(MPI_IN_PLACE, Minv.data(), Minv.size(), MPI_DOUBLE, MPI_SUM, _comm);
185 
186  Eigen::MatrixXd Wt = Minv.inverse()*Jmass.transpose();
187 
188  // Fill momenta.
189  Eigen::VectorXd momenta(3*vels.size());
190  for(size_t i = 0; i < vels.size(); ++i)
191  momenta.segment<3>(3*i) = mass[i]*vels[i];
192 
193  // Compute dot(w,p).
194  Eigen::VectorXd wdotp = Wt*momenta;
195 
196  // Reduce dot product across processors.
197  MPI_Allreduce(MPI_IN_PLACE, wdotp.data(), wdotp.size(), MPI_DOUBLE, MPI_SUM, _comm);
198 
199  // Compute d(wdotp)/dt second order backwards finite difference.
200  // Adding old force removes bias.
201  Eigen::VectorXd dwdotpdt = _unitconv*(1.5*wdotp - 2.0*_wdotp1 + 0.5*_wdotp2)/_timestep + _Fold;
202 
203  // If we are in bounds, sum force into running total.
204  if(coord != -1)
205  {
206  _F.segment(_ncv*coord, _ncv) += dwdotpdt;
207  ++_N[coord];
208  }
209 
210  // Reduce data across processors.
211  MPI_Allreduce(_F.data(), _Fworld.data(), _F.size(), MPI_DOUBLE, MPI_SUM, _world);
212  MPI_Allreduce(_N.data(), _Nworld.data(), _N.size(), MPI_INT, MPI_SUM, _world);
213 
214  // If we are in bounds, store the old summed force.
215  if(coord != -1)
216  _Fold = _Fworld.segment(_ncv*coord, _ncv)/std::max(_min, _Nworld[coord]);
217 
218  // Update finite difference time derivatives.
219  _wdotp2 = _wdotp1;
220  _wdotp1 = wdotp;
221 
222  // Write out data to file.
223  if(_iteration % _FBackupInterv == 0)
224  WriteData();
225 
226  // Calculate the bias from averaged F at current CV coordinates
227  CalcBiasForce(snapshot, cvs, coord);
228 
229  // Update the forces in snapshot by adding in the force bias from each
230  // CV to each atom based on the gradient of the CV.
231  for (size_t j = 0; j < forces.size(); ++j)
232  forces[j] += _biases[j];
233  }

243  void ABF::CalcBiasForce(const Snapshot* snapshot, const CVList& cvs, int coord)
244  {
245  // Reset the bias.
246  for(auto& b : _biases)
247  b.setZero();
248 
249  // Compute bias if within bounds
250  if(coord != -1)
251  {
252  for(int i = 0; i < _ncv; ++i)
253  {
254  auto& grad = cvs[i]->GetGradient();
255  for(size_t j = 0; j < _biases.size(); ++j)
256  _biases[j] -= _Fworld[_ncv*coord+i]*grad[j]/std::max(_min, _Nworld[coord]);
257  }
258  }
259  else
260  {
261  for(int i = 0; i < _ncv; ++i)
262  {
263  auto k = 0.;
264  auto x0 = 0.;
265 
266  if(cvs[i]->GetValue() < _restraint[i][0] && _restraint[i][2] > 0)
267  {
268  k = _restraint[i][2];
269  x0 = _restraint[i][0];
270  }
271  else if (cvs[i]->GetValue() > _restraint[i][1] && _restraint[i][2] > 0)
272  {
273  k = _restraint[i][2];
274  x0 = _restraint[i][1];
275  }
276 
277  auto& grad = cvs[i]->GetGradient();
278  for(size_t j = 0; j < _biases.size(); ++j)
279  _biases[j] -= (cvs[i]->GetValue() - x0)*k*grad[j];
280  }
281  }
282  }
