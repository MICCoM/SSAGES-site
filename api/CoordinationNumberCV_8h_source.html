45  double Evaluate(double rij, double& df) const
46  {
47  const auto xarg = (rij - d0_)/r0_;
48  const auto xn = std::pow(xarg, n_);
49  const auto xm = std::pow(xarg, m_);
50  const auto f = (1.-xn)/(1.-xm);
51 
52  df = f/(d0_-rij)*(n_*xn/(1.-xn)+m_*xm/(xm-1.));
53  return f;
54  }

164  void Evaluate(const Snapshot& snapshot) override
165  {
166  // Get local atom indices.
167  std::vector<int> idx1, idx2;
168  snapshot.GetLocalIndices(group1_, &idx1);
169  snapshot.GetLocalIndices(group2_, &idx2);
170 
171  // Get data from snapshot.
172  auto n = snapshot.GetNumAtoms();
173  auto& atomids = snapshot.GetAtomIDs();
174  auto& positions = snapshot.GetPositions();
175 
176  // Initialize gradient.
177  std::fill(grad_.begin(), grad_.end(), Vector3{0,0,0});
178  grad_.resize(n, Vector3{0,0,0});
179  boxgrad_ = Matrix3::Zero();
180 
181  // The nastiness begins. We essentially need to compute
182  // pairwise distances between the atoms. For now, let's
183  // gather the atomic coordinates of group2_
184  // on all relevant processors. Can be improved later.
185  auto& comm = snapshot.GetCommunicator();
186  std::vector<int> pcounts(comm.size(), 0), icounts(comm.size(), 0);
187  std::vector<int> pdispls(comm.size()+1, 0), idispls(comm.size()+1, 0);
188  pcounts[comm.rank()] = 3*idx2.size();
189  icounts[comm.rank()] = idx2.size();
190 
191  // Reduce counts.
192  MPI_Allreduce(MPI_IN_PLACE, pcounts.data(), pcounts.size(), MPI_INT, MPI_SUM, comm);
193  MPI_Allreduce(MPI_IN_PLACE, icounts.data(), icounts.size(), MPI_INT, MPI_SUM, comm);
194 
195  std::partial_sum(pcounts.begin(), pcounts.end(), pdispls.begin() + 1);
196  std::partial_sum(icounts.begin(), icounts.end(), idispls.begin() + 1);
197 
198  // Fill up mass and position vectors.
199  std::vector<double> pos(3*idx2.size(), 0);
200  std::vector<int> ids(idx2.size(), 0);
201  for(size_t i = 0; i < idx2.size(); ++i)
202  {
203  auto& idx = idx2[i];
204  auto& p = positions[idx];
205  pos[3*i+0] = p[0];
206  pos[3*i+1] = p[1];
207  pos[3*i+2] = p[2];
208  ids[i] = atomids[idx];
209  }
210 
211  // Create receiving vectors.
212  std::vector<double> gpos(pdispls.back(), 0);
213  std::vector<int> gids(idispls.back(), 0);
214 
215  // Gather data.
216  MPI_Allgatherv(pos.data(), pos.size(), MPI_DOUBLE, gpos.data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, comm);
217  MPI_Allgatherv(ids.data(), ids.size(), MPI_INT, gids.data(), icounts.data(), idispls.data(), MPI_INT, comm);
218 
219  val_ = 0;
220  double df = 0.;
221  for(auto& i : idx1)
222  {
223  auto& p = positions[i];
224  for(size_t j = 0; j < group2_.size(); ++j)
225  {
226  Vector3 rij = {p[0] - gpos[3*j], p[1] - gpos[3*j+1], p[2] - gpos[3*j+2]};
227  snapshot.ApplyMinimumImage(&rij);
228  auto r = rij.norm();
229  val_ += sf_.Evaluate(r, df);
230 
231  grad_[i] += df*rij/r;
232  boxgrad_ += df*rij/r*rij.transpose();
233 
234  auto lidx = snapshot.GetLocalIndex(gids[j]);
235  if(lidx != -1)
236  grad_[lidx] -= df*rij/r;
237  }
238  }
239 
240  // Reduce val.
241  MPI_Allreduce(MPI_IN_PLACE, &val_, 1, MPI_DOUBLE, MPI_SUM, comm);
242  }
