103  const auto& masses = snapshot.GetMasses();
104 
105  // Initialize gradient.
106  std::fill(grad_.begin(), grad_.end(), Vector3{0,0,0});
107  grad_.resize(n, Vector3{0,0,0});
108 
109  // Compute total and center of mass.
110  auto masstot = snapshot.TotalMass(idx);
111  Vector3 com = snapshot.CenterOfMass(idx, masstot);
112 
113  // Assign CV value.
114  switch(dim_)
115  {
116  case Dimension::x:
117  val_ = com[0];
118  break;
119  case Dimension::y:
120  val_ = com[1];
121  break;
122  case Dimension::z:
123  val_ = com[2];
124  break;
125  }
126 
127  // Assign gradient to appropriate atoms.
128  for(auto& id : idx)
129  {
130  grad_[id][0] = (dim_ == Dimension::x) ? masses[id]/masstot : 0;
131  grad_[id][1] = (dim_ == Dimension::y) ? masses[id]/masstot : 0;
132  grad_[id][2] = (dim_ == Dimension::z) ? masses[id]/masstot : 0;
133  }
