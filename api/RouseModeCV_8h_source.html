118  void Evaluate(const Snapshot& snapshot) override
119  {
120 
121  // Get data from snapshot.
122  auto ntot = snapshot.GetNumAtoms(); // total number of atoms
123  const auto& masses = snapshot.GetMasses(); // mass of each atom
124 
125  // Initialize working variables
126  double ppi_n = p_ * M_PI / N_; // constant
127  xp_.fill(0.0); // vectorial Rouse mode
128  r_.resize(N_); // position vector for beads in Rouse chain (unwrapped)
129  grad_.resize(ntot, Vector3{0,0,0}); // gradient set to 0.0 for all atoms
130  std::fill(grad_.begin(), grad_.end(), Vector3{0,0,0});
131 
132  // Iterate over all N_ atom groups and compute the center of mass for each
133  std::vector<Vector3> rcom; // vector of COM positions
134  for (size_t i = 0; i < N_; ++i) {
135  Label idi; // list of indices
136  snapshot.GetLocalIndices(groups_[i], &idi);
137  rcom.push_back(snapshot.CenterOfMass(idi,massg_[i])); // center of mass for group
138  }
139 
140  // Now compute differences vectors between the neighboring beads
141  // accumulate displacements to reconstruct unwrapped polymer chain
142  // for simplicity, we consider the first bead to be the reference position
143  // in all snapshots
144  r_[0] = rcom[0];
145  for (size_t i = 1; i< N_; ++i) {
146  Vector3 dri = snapshot.ApplyMinimumImage(rcom[i] - rcom[i-1]);
147  r_[i] = r_[i-1] + dri; // r_i = r_{i-1} + (r_i - r_{i-1})
148  }
149 
150  // Determine the value of the Rouse coordinate
151  // Xp(t) = 1/sqrt(N) * sum_{i=1}^{N} ri, p = 0
152  // Xp(t) = sqrt(2/N) * sum_{i=1}^{N} ri * cos[p*pi/N*(i-0.5)], p = 1,...,N-1
153  // Note: this solution is valid for homogeneous friction
154  xp_.fill(0.0);
155  for (size_t i = 0; i<N_; ++i) {
156  xp_ += r_[i]*cos ( ppi_n * (i+0.5) );
157  }
158  xp_ /= sqrt(N_);
159  if ( p_ != 0 ) xp_ *= sqrt(2.0) ;
160 
161  // Compute Rouse vector norm as the CV
162  // CV = sqrt(Xp*Xp), Xp = (Xp1,Xp2,Xp3)
163  val_ = xp_.norm();
164 
165  // Now perform gradient operation
166  // dCV/dxjd = (Xpd/CV)*(c/N)**0.5*sum_i=1^N cos[p*pi(i-0.5)/N] mj/Mi *delta_j({i})
167  // delta_j({i}) = 1, if j in {i}, 0 otherwise
168  Vector3 gradpcon = xp_ / sqrt(N_) / val_;
169  if ( p_ != 0) gradpcon *= sqrt(2.0); // (Xpd/CV)*(c/N)**0.5
170  for (size_t i = 0; i < N_; ++i) {
171  Label idi; // list of indices
172  snapshot.GetLocalIndices(groups_[i], &idi);
173  // go over each atom in the group and add to its gradient
174  // Note: performance tradeoff here. All gradient elements have a common factor of
175  // Xpd/CV*sqrt(c/N) = prefactor, with c = 2 if p != 0
176  // this could be factored out, but if ntot >> number of atoms in groups
177  // then it won't be worth it to post multiply all gradient terms...
178  // Could also go over loop again after to do the multiplication, but
179  // that is troublesome if atom ids appear in multiple groups for some reason
180  double cosval = cos(ppi_n*(i+0.5)) / massg_[i]; // cos[p*pi(i-0.5)/N] / Mi
181  for (auto& id : idi) {
182  grad_[id] += gradpcon* cosval * masses[id];
183  }
184  }
185
