111  void Meta::AddHill(const CVList& cvs)
112  {
113  int n = cvs.size();
114 
115  // Assume we have the same number of procs per walker.
116  int nwalkers = world_.size()/comm_.size();
117 
118  // We need to exchange CV values across the walkers
119  // and to each proc on a walker.
120  std::vector<double> cvals(n*nwalkers, 0);
121 
122  if(comm_.rank() == 0)
123  {
124  for(auto i = 0, j = world_.rank()/comm_.size()*n; i < n; ++i,++j)
125  cvals[j] = cvs[i]->GetValue();
126  }
127 
128  // Reduce across all processors and add hills.
129  MPI_Allreduce(MPI_IN_PLACE, cvals.data(), n*nwalkers, MPI_DOUBLE, MPI_SUM, world_);
130 
131  for(int i = 0; i < n*nwalkers; i += n)
132  {
133  std::vector<double> cval(cvals.begin() + i, cvals.begin() + i + n);
134  hills_.emplace_back(cval, widths_, height_);
135 
136  // Write hill to file.
137  if(world_.rank() == 0)
138  PrintHill(hills_.back());
139  }
140  }

155  void Meta::CalcBiasForce(const CVList& cvs)
156  {
157  // Reset bias and derivatives.
158  double bias = 0.;
159  auto n = cvs.size();
160 
161  // Reset vectors.
162  std::fill(derivatives_.begin(), derivatives_.end(), 0);
163 
164  // Loop through hills and calculate the bias force.
165  for(auto& hill : hills_)
166  {
167  auto tbias = 1.;
168  std::fill(tder_.begin(), tder_.end(), 1.0);
169  std::fill(dx_.begin(), dx_.end(), 1.0);
170 
171  for(size_t i = 0; i < n; ++i)
172  {
173  dx_[i] = cvs[i]->GetDifference(hill.center[i]);
174  tbias *= gaussian(dx_[i], hill.width[i]);
175  }
176 
177  for(size_t i = 0; i < n; ++i)
178  for(size_t j = 0; j < n; ++j)
179  {
180  if(j != i)
181  tder_[i] *= gaussian(dx_[j], hill.width[j]);
182  else
183  tder_[i] *= gaussianDerv(dx_[j], hill.width[j]);
184  }
185 
186  bias += height_ * tbias;
187  for(size_t i = 0; i < n; ++i)
188  derivatives_[i] += height_*tder_[i];
189  }
190  }
