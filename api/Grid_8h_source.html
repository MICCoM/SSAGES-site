146  static Grid<T>* BuildGrid(const Json::Value& json, const std::string& path)
147  {
148  Json::ObjectRequirement validator;
149  Json::Value schema;
150  Json::Reader reader;
151 
152  reader.parse(JsonSchema::grid, schema);
153  validator.Parse(schema, path);
154 
155  // Validate inputs.
156  validator.Validate(json, path);
157  if (validator.HasErrors()) {
158  throw BuildException(validator.GetErrors());
159  }
160 
161  // Read in Lower Grid edges.
162  std::vector<double> lower;
163  for (auto &lv : json["lower"]) {
164  lower.push_back(lv.asDouble());
165  }
166 
167  size_t dimension = lower.size();
168 
169  // Read in upper grid edges.
170  std::vector<double> upper;
171  for (auto &uv : json["upper"]) {
172  upper.push_back(uv.asDouble());
173  }
174 
175  if (upper.size() != dimension) {
176  throw BuildException({"Number of upper values does not match "
177  "number of lower values!"});
178  }
179 
180  // Read in number of points.
181  std::vector<int> number_points;
182  for (auto &np : json["number_points"]) {
183  number_points.push_back(np.asInt());
184  }
185 
186  if (number_points.size() != dimension) {
187  throw BuildException({"Arrays \"lower\" and \"number_points\" do "
188  "not have the same size!"});
189  }
190 
191  // Read in periodicity.
192  std::vector<bool> isPeriodic;
193  for (auto &periodic : json["periodic"]) {
194  isPeriodic.push_back(periodic.asBool());
195  }
196 
197  if (isPeriodic.size() == 0) {
198  isPeriodic = std::vector<bool>(dimension, false);
199  } else if (isPeriodic.size() != dimension) {
200  throw BuildException({"Arrays \"lower\" and \"periodic\" do not "
201  "have the same size!"});
202  }
203 
204  // Construct the grid.
205  Grid<T>* grid = new Grid(number_points, lower, upper, isPeriodic);
206 
207  return grid;
208  }
