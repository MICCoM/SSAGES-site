430  void ForwardFlux::ComputeCommittorProbability(Snapshot *snapshot){
431 
432  // two dim vectors, first dim is lambda, second is N
433  // this counts the number of atempts from this config eventually reached A (for nA) or B (for nB)
434  // used to compute _pB via: _pB = nB / (nA + nB)
435  std::vector<std::vector<unsigned int>> nA;
436  std::vector<std::vector<unsigned int>> nB;
437  nA.resize(_ninterfaces);
438  nB.resize(_ninterfaces);
439  for (unsigned int i=0; i<_ninterfaces;i++){
440  nA[i].resize(_N[i],0);
441  nB[i].resize(_N[i],0);
442  }
443 
444 
445  //Snapshot* snapshot_empty;
446  FFSConfigID ffsconfig;
447 
448  //populate nB
449  unsigned int nsuccess = _N[_ninterfaces-1]; //note -1
450  for(unsigned int i = 0; i < nsuccess ; i++){
451  ffsconfig.l = _ninterfaces-1;
452  ffsconfig.n = i;
453  ffsconfig.a = 0;
454  bool flag = true;
455  //recursively trace successful path and update all the configs it came from
456  while (flag){
457  if (ffsconfig.l == 0){ flag = false;}
458 
459  //this is just to populate ffsconfig.{lprev,nprev,aprev}
460  ReadFFSConfiguration(snapshot,ffsconfig,true);
461 
462  //update nB
463  nB[ffsconfig.l][ffsconfig.n]++;
464 
465  ffsconfig.l = ffsconfig.lprev;
466  ffsconfig.n = ffsconfig.nprev;
467  ffsconfig.a = ffsconfig.aprev;
468  }
469  }
470 
471  //now populate nA (very similar to nB)
472  unsigned int nfail = _nfailure_total;
473  for(unsigned int i = 0; i < nfail ; i++){
474  ffsconfig.l = 0; //only fail at lambda0 in absence of pruning
475  ffsconfig.n = i;
476  ffsconfig.a = 0;
477  bool flag = true;
478  //this is just to populate ffsconfig.{lprev,nprev,aprev}
479  ReadFFSConfiguration(snapshot,ffsconfig,false);
480 
481  //recursively trace successful path and update all the configs it came from
482  while (flag){
483  if ((ffsconfig.l == 0) && (ffsconfig.lprev==0)){ flag = false;}
484 
485  //update nA
486  nA[ffsconfig.l][ffsconfig.n]++;
487 
488  ReadFFSConfiguration(snapshot,ffsconfig,true);
489 
490  ffsconfig.l = ffsconfig.lprev;
491  ffsconfig.n = ffsconfig.nprev;
492  ffsconfig.a = ffsconfig.aprev;
493  }
494  }
495 
496  //Compute _pB
497  _pB.resize(_ninterfaces);
498  unsigned int Nmax = 0;
499  for(unsigned int i = 0; i<_ninterfaces ; i++){
500  _pB[i].resize(_N[i],0);
501  for(unsigned int j = 0; j<_N[i] ; j++){
502  int denom = nA[i][j] + nB[i][j];
503  if (denom != 0){
504  _pB[i][j] = (double)nB[i][j] / denom;
505  }
506  }
507  if (_N[i] > Nmax){ Nmax = _N[i];}
508  }
509 
510  //print pB
511  //rows are lambda, cols are n. Thats why its looks complicated
512  std::ofstream file;
513  std::string filename;
514  filename = _output_directory +"/commitor_probabilities.dat";
515  file.open(filename.c_str());

517 
518  for(unsigned int i = 0; i<Nmax ; i++){
519  for(unsigned int j = 0; j<_ninterfaces ; j++){
520  if (i < _N[j]){
521  file << _pB[j][i] << " ";
