59  void ForwardFlux::PostIntegration(Snapshot* snapshot, const CVList& cvs)
60  {
61  switch(_restart)
62  {
63  case NEW:
64  {
65  SetUpNewLibrary(snapshot, cvs);
66  return;
67  }
68  case LIBRARY:
69  {
70  if(_world.rank() == 0 && _currentstartingpoint != 0)
71  {
72  for(auto& value : _successes)
73  _resultsfile<<value<<" ";
74  _resultsfile<<"\n";
75  }
76 
77  mpi::all_reduce(_world, _totalcontents, _globalcontents, std::plus<std::string>());
78  //Close local and global files
79  if(_world.rank() == 0)
80  {
81  _indexfile<<_globalcontents<<std::endl;
82  }
83  _totalcontents += _indexcontents;
84  _indexcontents = "";
85 
86  _currentshot = 0;
87 
88  std::fill(_localsuccesses.begin(),_localsuccesses.end(),0);
89  std::fill(_successes.begin(),_successes.end(),0);
90 
91  _currentnode = 1;
92 
93  std::vector<std::vector<std::string> > tmp;
94  if(!(ExtractInterfaceIndices(0, _librarycontents, tmp)))
95  {
96  std::cout<< "Could not locate any files";
97  std::cout<< " at first interface!"<<std::endl;
98  std::cout<< _librarycontents<<std::endl;
99  PostSimulation(snapshot, cvs);
100  _world.abort(0);
101  }
102 
103  if(_currentstartingpoint >= tmp.size())
104  {
105  if(_world.rank()==0)
106  std::cout<<"Ending Simulation..."<<std::endl;
107  PostSimulation(snapshot, cvs);
108  _world.abort(0);
109  }
110 
111  _shootingconfigfile = tmp[_currentstartingpoint][1];
112  if(_world.rank() == 0)
113  _currentconfig = GetFileContents(_shootingconfigfile.c_str());
114  mpi::broadcast(_world, _currentconfig, 0);
115 
116  ReadConfiguration(snapshot, _currentconfig);
117  _restart = NONE;
118  _currentstartingpoint++;
119  return;
120  }
121  case NEWCONFIG:
122  {
123  _currentshot = 0;
124  mpi::all_reduce(_world, _indexcontents, _globalcontents, std::plus<std::string>());
125 
126  _shootingconfigfile = PickConfiguration(_currentnode, _globalcontents);
127 
128  if(_world.rank() == 0)
129  _currentconfig = GetFileContents(_shootingconfigfile.c_str());
130 
131  mpi::broadcast(_world, _currentconfig, 0);
132  ReadConfiguration(snapshot, _currentconfig);
133  _restart = NONE;
134 
135  return;
136  }
137  default:
138  {
139  break;
140  }
141  }
142 
143  // Locate the interface you are at and check if:
144  // Returned to origin or at next interface
145  int atinter = AtInterface(cvs);
146  if(atinter == _currentnode + 1 || atinter == 0)
147  {
148  _currentnode++;
149  // Check if you made it to the next one!
150  if(atinter == _currentnode)
151  {
152  auto& ID = snapshot->GetSnapshotID();
153  ID = "dump_"+std::to_string(_currentnode)+"_"+std::to_string(_currenthash)+".dump";
154  _currenthash++;
155  if(_comm.rank()==0)
156  {
157  WriteConfiguration(snapshot);
158  _localsuccesses[_currentnode]++;
159  }
160  }
161 
162  if(_currentshot < _numshots)
163  {
164  _currentshot++;
165  ReadConfiguration(snapshot, _currentconfig);
166  _currentnode--;
167  _iteration++;
168  return;
169  }
170 
171  mpi::all_reduce(_world, _localsuccesses[_currentnode], _successes[_currentnode], std::plus<int>());
172  if(_successes[_currentnode] > 0)
173  {
174  _restart = NEWCONFIG;
175 
176  // If you have reached the final interface you are "done" with that path
177  if(_currentnode >= _centers.size()-1)
178  {
179  std::cout<< "Found finishing configuration! "<<std::endl;
180  _restart = LIBRARY;
181  }
182  }
183  else
184  _restart = LIBRARY;
185 
186  _iteration++;
187  }
188  }

205  void ForwardFlux::SetUpNewLibrary(Snapshot* snapshot, const CVList& cvs)
206  {
207  // Get CV values check if at next interface, if so store configuration
208  int interface = AtInterface(cvs);
209  _shootingconfigfile = "Origin";
210 
211  // Flux out of A
212  if(interface == 1 && _currentnode == 0)
213  {
214  auto& ID = snapshot->GetSnapshotID();

216  _currenthash++;

218  {
219  WriteConfiguration(snapshot);
220  _localsuccesses[_currentnode]++;
221  }
222 
223  _fluxout++;
224  }
225  // Flux back in towards A
226  else if(interface == 0 && _currentnode == 1)
227  _fluxin++;
228

230 
231  std::vector<std::vector<std::string> > TempLibrary;
232  ExtractInterfaceIndices(0, _indexcontents, TempLibrary);
233

274  bool ForwardFlux::ExtractInterfaceIndices(int interface, const std::string& contents,
275  std::vector<std::vector<std::string> >& InterfaceIndices)
276  {
277  //Extract configuration indices for a given interface int
278  std::istringstream f(contents);
279  std::string line;
280  while (std::getline(f, line))
281  {
282  std::string buf; // Have a buffer string
283  std::stringstream ss(line); // Insert the string into a stream
284  std::vector<std::string> tokens; // Create vector to hold our words
285 
286  while (ss >> buf)

294  return false;

299  void ForwardFlux::ReadConfiguration(Snapshot* snapshot, const std::string& FileContents)
300  {
301  auto& positions = snapshot->GetPositions();
302  auto& velocities = snapshot->GetVelocities();
303  auto& atomID = snapshot->GetAtomIDs();
304  auto& forces = snapshot->GetForces();
305  auto& ID = snapshot->GetSnapshotID();
306 
307  ID = _shootingconfigfile;
308 
309  //Extract currentconfig information
310  std::istringstream f(FileContents);
311  std::string line;
312  while (std::getline(f, line))
313  {
314  int atomindex = -1;
315  std::string buf; // Have a buffer string
316  std::stringstream ss(line); // Insert the string into a stream
317  std::vector<std::string> tokens; // Create vector to hold our words
318 
319  while (ss >> buf)
320  tokens.push_back(buf);
321 
322  if(tokens.size() != 7)
323  {
324  std::cout<<"error, incorrect line format in "<<_shootingconfigfile<<" on line: "<<std::endl;
325  std::cout<<line<<std::endl;
326  _world.abort(-1);
327  }
328 
329  for(size_t i=0; i < atomID.size(); i++)
330  {
331  if(atomID[i] == std::stoi(tokens[0]))
332  atomindex = i;
333  }
334 
335  if(atomindex < 0)
336  {
337  std::cout<<"error, could not locate atomID "<<tokens[0]<<" from dumpfile"<<std::endl;
338  _world.abort(-1);
339  }
340 
341  positions[atomindex][0] = std::stod(tokens[1]);
342  positions[atomindex][1] = std::stod(tokens[2]);
343  positions[atomindex][2] = std::stod(tokens[3]);
344  velocities[atomindex][0] = std::stod(tokens[4]);
345  velocities[atomindex][1] = std::stod(tokens[5]);
346  velocities[atomindex][2] = std::stod(tokens[6]);
347 
348  for(auto& force : forces)
349  force.setZero();
350  }
