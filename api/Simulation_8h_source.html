191  bool BuildDriver(const Json::Value& json, const std::string& path)
192  {
193  ArrayRequirement validator;
194  Value schema;
195  Value JsonDrivers;
196  Value JsonDriver;
197  Reader reader;
198 
199  bool success_build = true;
200 
201  reader.parse(JsonSchema::driver, schema);
202  validator.Parse(schema, path);
203 
204  JsonDrivers = json.get("driver",Json::arrayValue);
205  try
206  {
207  // Validate inputs.
208  validator.Validate(JsonDrivers, path);
209  if(validator.HasErrors())
210  {
211  throw BuildException(validator.GetErrors());
212  }
213  } catch(BuildException& e) {
214  DumpErrorsToConsole(e.GetErrors(), notw_);
215  return false;
216  } catch(std::exception& e) {
217  DumpErrorsToConsole({e.what()}, notw_);
218  return false;
219  }
220 
221  // Determine walker ID using JSON file and split
222  // communicator accordingly.
223  int i = 0;
224  int totalproc = 0;
225  int wid = 0;
226  for(auto& m : JsonDrivers)
227  {
228  int currentnumproc = m.get("number processors", 1).asInt();
229 
230  if ((int)world_.rank() >= totalproc && (int)world_.rank() < currentnumproc + totalproc)
231  {
232  wid = i;
233  JsonDriver = m;
234  }
235 
236  totalproc += currentnumproc;
237  i++;
238  }
239 
240  nwalkers_ = JsonDrivers.size();
241  comm_ = world_.split(wid);
242 
243  if(world_.size() != totalproc)
244  {
245  if(world_.rank() == 0)
246  {
247  std::cerr << "The number of processors for each driver (walker) must sum "
248  << "to the total processors for the mpi call." << std::endl;
249  std::cerr<<world_.size()<<" vs "<<totalproc<<std::endl;
250  }
251  return false;
252  }
253 
254  // Get the engine.
255  MDEngine_ = JsonDriver.get("type", "none").asString();
256 
257  // Use input from JSON to determine MDEngine of choice as well as other parameters
258  #ifdef ENABLE_LAMMPS
259  if(MDEngine_ == "LAMMPS")
260  {
261  LammpsDriver* en = new LammpsDriver(world_, comm_, wid);
262 
263  if(!(MDDriver_ = static_cast<Driver*>(en)))
264  {
265  std::cerr << "Unable to dynamic cast engine on node "<<world_.rank()<<" Error occurred" << std::endl;
266  success_build = false;
267  }
268  }
269  else
270  #endif
271  #ifdef ENABLE_GROMACS
272  if(MDEngine_ == "Gromacs")
273  {
274  GromacsDriver* en = new GromacsDriver(world_, comm_, wid);
275  if(!(MDDriver_ = static_cast<Driver*>(en)))
276  {
277  std::cout << "Unable to static cast engine on node " << world_.rank() << std::endl;
278  success_build = false;
279  }
280  }
281  else
282  #endif
283  {
284  DumpErrorsToConsole({"Unknown MD Engine [" + MDEngine_ + "] specified."},notw_);
285  success_build = false;
286  }
287 
288  if (!success_build) {
289  // If building the MDEngine failed, don't bother about the rest.
290  return success_build;
291  }
292 
293  // Build the driver
294  try{
295  MDDriver_->BuildDriver(JsonDriver, path + "/" + std::to_string(wid));
296  } catch(BuildException& e) {
297  DumpErrorsToConsole(e.GetErrors(), notw_);
298  success_build = false;
299  } catch(std::exception& e) {
300  DumpErrorsToConsole({e.what()}, notw_);
301  success_build = false;
302  }
303 
304  // Build the CVs
305  if(JsonDriver.isMember("CVs"))
306  {
307  if(!BuildCVs(JsonDriver, "#/CVs"))
308  success_build = false;
309  }
310  else
311  {
312  if(!BuildCVs(json,"#/CVs"))
313  success_build = false;
314  }
315 
316  // Build the method
317  if (JsonDriver.isMember("method"))
318  {
319  if(!BuildMethod(JsonDriver, "#/Methods"))
320  success_build = false;
321  }
322  else if (json.isMember("method"))
323  {
324  if(!BuildMethod(json,"#/Methods"))
325  success_build = false;
326  }
327 
328  if(JsonDriver.isMember("observers"))
329  {
330  if(!BuildObservers(JsonDriver))
331  success_build = false;
332  }
333  else if (json.isMember("observers"))
334  {
335  if(!BuildObservers(json))
336  success_build = false;
337  }
338 
339  // Build the constraints if they exist
340  if(JsonDriver.isMember("constraints"))
341  {
342  if(!BuildConstraints(JsonDriver, "#/Constraints"))
343  success_build = false;
344  }
345  else if (json.isMember("constraints"))
346  {
347  if(!BuildConstraints(json, "#/Constraints"))
348  success_build = false;
349  }
350 
351  if(comm_.rank()==0)
352  {
353  if(success_build)
354  std::cout << std::setw(notw_) << std::right << "\033[32mMDEngine " << wid << " pass!\033[0m\n";
355  else
356  std::cout << std::setw(notw_) << std::right << "\033[32mMDEngine " << wid << " FAIL!\033[0m\n";
357  }
358 
359  return success_build;
360  }
