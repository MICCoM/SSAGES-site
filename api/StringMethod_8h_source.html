156  void StringReparam(double alpha_star)
157  {
158  std::vector<double> alpha_star_vector(numnodes_,0.0);
159 
160  //Reparameterization
161  //Alpha star is the uneven mesh, approximated as linear distance between points
162  if(comm_.rank()==0)
163  alpha_star_vector[mpiid_] = mpiid_ == 0 ? 0 : alpha_star;
164 
165  //Gather each alpha_star into a vector
166  MPI_Allreduce(MPI_IN_PLACE, &alpha_star_vector[0], numnodes_, MPI_DOUBLE, MPI_SUM, world_);
167 
168  for(size_t i = 1; i < alpha_star_vector.size(); i++)
169  alpha_star_vector[i] += alpha_star_vector[i-1];
170 
171  for(size_t i = 1; i < alpha_star_vector.size(); i++)
172  alpha_star_vector[i] /= alpha_star_vector[numnodes_ - 1];
173 
174  tk::spline spl; //Cubic spline interpolation
175 
176  for(size_t i = 0; i < centers_.size(); i++)
177  {
178  std::vector<double> cvs_new(numnodes_, 0.0);
179 
180  if(comm_.rank() == 0)
181  cvs_new[mpiid_] = centers_[i];
182 
183  MPI_Allreduce(MPI_IN_PLACE, &cvs_new[0], numnodes_, MPI_DOUBLE, MPI_SUM, world_);
184 
185  spl.set_points(alpha_star_vector, cvs_new);
186  centers_[i] = spl(mpiid_/(numnodes_ - 1.0));
187  }
188  }
