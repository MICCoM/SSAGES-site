485  Vector3 CenterOfMass(const Label& indices, double mtot) const
486  {
487  // Store coorinates and masses in vectors to gather.
488  std::vector<double> pos, mass, gpos, gmass;
489  std::vector<int> pcounts(comm_.size(), 0), mcounts(comm_.size(), 0);
490  std::vector<int> pdispls(comm_.size()+1, 0), mdispls(comm_.size()+1, 0);
491 
492  pcounts[comm_.rank()] = 3*indices.size();
493  mcounts[comm_.rank()] = indices.size();
494 
495  // Reduce counts.
496  MPI_Allreduce(MPI_IN_PLACE, pcounts.data(), pcounts.size(), MPI_INT, MPI_SUM, comm_);
497  MPI_Allreduce(MPI_IN_PLACE, mcounts.data(), mcounts.size(), MPI_INT, MPI_SUM, comm_);
498 
499  // Compute displacements.
500  std::partial_sum(pcounts.begin(), pcounts.end(), pdispls.begin() + 1);
501  std::partial_sum(mcounts.begin(), mcounts.end(), mdispls.begin() + 1);
502 
503  // Fill up mass and position vectors.
504  for(auto& idx : indices)
505  {
506  auto& p = positions_[idx];
507  pos.push_back(p[0]);
508  pos.push_back(p[1]);
509  pos.push_back(p[2]);
510  mass.push_back(masses_[idx]);
511  }
512 
513  // Re-size receiving vectors.
514  gpos.resize(pdispls.back(), 0);
515  gmass.resize(mdispls.back(), 0);
516 
517  // All-gather data.
518  MPI_Allgatherv(pos.data(), pos.size(), MPI_DOUBLE, gpos.data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, comm_);
519  MPI_Allgatherv(mass.data(), mass.size(), MPI_DOUBLE, gmass.data(), mcounts.data(), mdispls.data(), MPI_DOUBLE, comm_);
520 
521  // Loop through atoms and compute mass weighted sum.
522  // We march linearly through list and find nearest image
523  // to each successive particle to properly unwrap object.
524  Vector3 ppos = {gpos[0], gpos[1], gpos[2]}; // Previous unwrapped position.
525  Vector3 cpos = ppos;
526  Vector3 xcm = gmass[0]*cpos;
527 
528  for(size_t i = 1, j = 3; i < gmass.size(); ++i, j += 3)
529  {
530  cpos = {gpos[j], gpos[j+1], gpos[j+2]};
531  cpos = ApplyMinimumImage(cpos - ppos) + ppos;
532  xcm += gmass[i]*cpos;
533  ppos = cpos;
534  }
535 
536  return xcm/mtot;
537  }
