455  Vector3 CenterOfMass(const Label& indices, double mtot) const
456  {
457  // Store coorinates and masses in vectors to gather.
458  std::vector<double> pos, mass, gpos, gmass;
459  std::vector<int> pcounts(comm_.size(), 0), mcounts(comm_.size(), 0);
460  std::vector<int> pdispls(comm_.size()+1, 0), mdispls(comm_.size()+1, 0);
461 
462  pcounts[comm_.rank()] = 3*indices.size();
463  mcounts[comm_.rank()] = indices.size();
464 
465  // Reduce counts.
466  MPI_Allreduce(MPI_IN_PLACE, pcounts.data(), pcounts.size(), MPI_INT, MPI_SUM, comm_);
467  MPI_Allreduce(MPI_IN_PLACE, mcounts.data(), mcounts.size(), MPI_INT, MPI_SUM, comm_);
468 
469  // Compute displacements.
470  std::partial_sum(pcounts.begin(), pcounts.end(), pdispls.begin() + 1);
471  std::partial_sum(mcounts.begin(), mcounts.end(), mdispls.begin() + 1);
472 
473  // Fill up mass and position vectors.
474  for(auto& idx : indices)
475  {
476  auto& p = positions_[idx];
477  pos.push_back(p[0]);
478  pos.push_back(p[1]);
479  pos.push_back(p[2]);
480  mass.push_back(masses_[idx]);
481  }
482 
483  // Re-size receiving vectors.
484  gpos.resize(pdispls.back(), 0);
485  gmass.resize(mdispls.back(), 0);
486 
487  // All-gather data.
488  MPI_Allgatherv(pos.data(), pos.size(), MPI_DOUBLE, gpos.data(), pcounts.data(), pdispls.data(), MPI_DOUBLE, comm_);
489  MPI_Allgatherv(mass.data(), mass.size(), MPI_DOUBLE, gmass.data(), mcounts.data(), mdispls.data(), MPI_DOUBLE, comm_);
490 
491  // Loop through atoms and compute mass weighted sum.
492  // We march linearly through list and find nearest image
493  // to each successive particle to properly unwrap object.
494  Vector3 ppos = {gpos[0], gpos[1], gpos[2]}; // Previous unwrapped position.
495  Vector3 cpos = ppos;
496  Vector3 xcm = gmass[0]*cpos;
497 
498  for(size_t i = 1, j = 3; i < gmass.size(); ++i, j += 3)
499  {
500  cpos = {gpos[j], gpos[j+1], gpos[j+2]};
501  cpos = ApplyMinimumImage(cpos - ppos) + ppos;
502  xcm += gmass[i]*cpos;
503  ppos = cpos;
504  }
505 
506  return xcm/mtot;
507  }
