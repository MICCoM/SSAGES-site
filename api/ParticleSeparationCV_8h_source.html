85  auto n1 = group1_.size(), n2 = group2_.size();
86 
87  // Make sure atom ID's are on at least one processor.
88  std::vector<int> found1(n1, 0), found2(n2, 0);
89  for(size_t i = 0; i < n1; ++i)
90  {
91  if(snapshot.GetLocalIndex(group1_[i]) != -1)
92  found1[i] = 1;
93  }
94 
95  for(size_t i = 0; i < n2; ++i)
96  {
97  if(snapshot.GetLocalIndex(group2_[i]) != -1)
98  found2[i] = 1;
99  }
100 
101  MPI_Allreduce(MPI_IN_PLACE, found1.data(), n1, MPI_INT, MPI_SUM, snapshot.GetCommunicator());
102  MPI_Allreduce(MPI_IN_PLACE, found2.data(), n2, MPI_INT, MPI_SUM, snapshot.GetCommunicator());
103 
104  unsigned ntot1 = std::accumulate(found1.begin(), found1.end(), 0, std::plus<int>());
105  unsigned ntot2 = std::accumulate(found2.begin(), found2.end(), 0, std::plus<int>());
106  if(ntot1 != n1)
107  throw BuildException({

127  void Evaluate(const Snapshot& snapshot) override
128  {
129  // Get local atom indices.
130  std::vector<int> idx1, idx2;
131  snapshot.GetLocalIndices(group1_, &idx1);
132  snapshot.GetLocalIndices(group2_, &idx2);
133 
134  // Get data from snapshot.
135  auto n = snapshot.GetNumAtoms();
136  const auto& masses = snapshot.GetMasses();
137 
138  // Initialize gradient.
139  std::fill(grad_.begin(), grad_.end(), Vector3{0,0,0});
140  grad_.resize(n, Vector3{0,0,0});
141  boxgrad_ = Matrix3::Zero();
142 
143  // Get centers of mass.
144  auto mtot1 = snapshot.TotalMass(idx1);
145  auto mtot2 = snapshot.TotalMass(idx2);
146  Vector3 com1 = snapshot.CenterOfMass(idx1, mtot1);
147  Vector3 com2 = snapshot.CenterOfMass(idx2, mtot2);
148 
149  // Account for pbc.
150  Vector3 rij = snapshot.ApplyMinimumImage(com1 - com2).cwiseProduct(dim_.cast<double>());
151 
152  // Compute gradient.
153  val_ = rij.norm();
154 
155  for(auto& id : idx1)
156  {
157  grad_[id] = rij/val_*masses[id]/mtot1;
158  boxgrad_ += grad_[id]*rij.transpose();
159  }
160 
161  for(auto& id : idx2)
162  grad_[id] = -rij/val_*masses[id]/mtot2;
163  }
