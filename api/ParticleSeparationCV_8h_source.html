107  {

113  // Get data from snapshot.
114  auto n = snapshot.GetNumAtoms();
115  const auto& masses = snapshot.GetMasses();
116 
117  // Initialize gradient.
118  std::fill(_grad.begin(), _grad.end(), Vector3{0,0,0});
119  _grad.resize(n, Vector3{0,0,0});
120 
121  // Get centers of mass.
122  auto mtot1 = snapshot.TotalMass(idx1);
123  auto mtot2 = snapshot.TotalMass(idx2);
124  Vector3 com1 = snapshot.CenterOfMass(idx1, mtot1);
125  Vector3 com2 = snapshot.CenterOfMass(idx2, mtot2);
126 
127  // Account for pbc.
128  Vector3 rij = snapshot.ApplyMinimumImage(com1 - com2);
129 
130  // Compute gradient.
131  _val = rij.norm();
132 
133  for(auto& id : idx1)
134  _grad[id] = rij/_val*masses[id]/mtot1;
135 
136  for(auto& id : idx2)
137  _grad[id] = -rij/_val*masses[id]/mtot2;
